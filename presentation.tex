\sectionsubtitle{Polynomial fitting for gradient estimation}
\section{Background}

\begin{frame}{Local Polynomial Gradient Estimation}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      Model local intensity with a 2D Taylor expansion of order~$d$:
      \begin{equation*}
        f(x_i, y_i) \approx f^0 + f_x x_i + f_y y_i + \tfrac{f_{xx}}{2} x_i^2 + \cdots
      \end{equation*}
      $M = \tfrac{(d{+}1)(d{+}2)}{2}$ unknowns \quad ($d{=}4 \to M{=}15$)

      \vspace{0.4em}
      \begin{itemize}
        \item Gather $N_p$ circular neighbors $\;\to\;$ system $\mathbf{A}\mathbf{z} = \mathbf{b}$
        \item Solve via pseudoinverse: $\;\hat{\mathbf{z}} = \mathbf{A}^+ \mathbf{b}$
        \item Normal derivative $=$ row~1 of $\mathbf{A}^+$ $\cdot$ intensities
      \end{itemize}

      \vspace{0.4em}
      \begin{block}{Key Property}
        $\mathbf{A}^+$ depends only on geometry and $\theta$
        $\;\to\;$ \textbf{precompute once}, reuse per pixel.
      \end{block}
    \end{column}
    \begin{column}{0.43\textwidth}
      \centering
      \vspace{-0.3em}
      \includegraphics[width=\textwidth]{fig3_stencil_orientations.png}

      \vspace{0.3em}
      {\scriptsize Fused stencil weights at 6 orientations.\\
      Red = positive, blue = negative weight.}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Line Extension for Robustness}
  Evaluate the point filter at $(2m{+}1)$ virtual positions along each candidate edge direction:
  \begin{equation*}
    R_k = \sum_{j=-m}^{m} w_j \cdot \hat{f}_x^{(j,k)},
    \qquad w_j = \exp\!\bigl(-\tfrac{j^2}{2\sigma^2}\bigr),\quad \sigma = m/2
  \end{equation*}

  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      \textbf{Benefits:}
      \begin{itemize}
        \item Wider spatial context $\;\to\;$ noise robustness
        \item Coherence along the edge tangent $\;\to\;$ bridges gaps
      \end{itemize}

      \vspace{0.5em}
      \textbf{Problem:}
      \begin{itemize}
        \item $(2m{+}1)$ independent gather-matmul ops per orientation per pixel
        \item Massive memory: $L \times N_p$ intermediate values
        \item Many pixels are read multiple times (overlap)
      \end{itemize}
    \end{column}
    \begin{column}{0.42\textwidth}
      \centering
      \vspace{0.5em}
      \includegraphics[width=0.85\textwidth]{fig1_neighborhood_comparison.png}

      \vspace{0.3em}
      {\scriptsize Panel B shows 5 overlapping\\neighborhoods along the line}
    \end{column}
  \end{columns}
\end{frame}

\sectionsubtitle{Per-pixel operation count for the naive approach}
\section{Naive Line Filter Cost}

\begin{frame}[fragile]{Naive Line Filter: Pseudocode}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \begin{lstlisting}[language=Python, escapeinside={(*}{*)}]
for (*$\theta$*) in orientations:  # (*$N_s$*) = 18
  for j = -m to m:        # L = 15
    for i = 1 to (*$N_p$*):     # (*$N_p$*) = 100
      READ pixel(j, i)    # 1 mem read
      MUL by (*$p_i^{(k)}$*)     # 1 multiply
    SUM (*$N_p$*) values         # 99 adds
  MUL by Gaussian (*$w_j$*)    # 15 muls
  SUM over line           # 14 adds
  ABS((*$R_\theta$*))
pick max over (*$\theta$*)
      \end{lstlisting}
    \end{column}
    \begin{column}{0.45\textwidth}
      \textbf{Per orientation} ($m{=}7,\ N_p{=}100$):
      \begin{itemize}
        \item Memory reads: $15 \times 100 = \mathbf{1{,}500}$
        \item Multiplies: $15 \times 100 + 15 = \mathbf{1{,}515}$
        \item Additions: $15 \times 99 + 14 = \mathbf{1{,}499}$
      \end{itemize}

      \vspace{0.6em}
      \textbf{Per pixel} (18 orientations):
      \begin{itemize}
        \item Reads: \textbf{27,000}
        \item Multiplies: \textbf{27,270}
        \item Additions: \textbf{26,982}
        \item[\color{UofSCGarnet}$\blacksquare$] \textbf{Total: $\sim$81,252 ops}
      \end{itemize}
    \end{column}
  \end{columns}

  \vspace{0.3em}
  \begin{alertblock}{Complexity}
    $\mathcal{O}\!\bigl(N_s \times (2m{+}1) \times N_p\bigr)$ per pixel $\;=\; \mathcal{O}(N_s \cdot L \cdot N_p)$
  \end{alertblock}
\end{frame}

\begin{frame}{Naive LF: Computation Flow}
  \begin{center}
    \includegraphics[width=0.92\textwidth]{fig2_computation_flow.png}
  \end{center}

  \vspace{0.3em}
  \begin{itemize}
    \item \textbf{Top:} Naive LF requires nested loops --- 8 sequential stages per pixel
    \item \textbf{Bottom:} Fused ASF collapses to 6 stages, with the inner 5 replaced by one gather + dot product
    \item Memory: naive allocates \textbf{6.2~GB} of intermediates vs.\ \textbf{20~MB} for ASF
  \end{itemize}
\end{frame}

\sectionsubtitle{Collapsing the pipeline into a single dot product}
\section{Fused Stencil Formulation}

\begin{frame}{Algebraic Collapse}
  The line-extended response is a \textbf{linear operation} on pixel intensities:
  \begin{equation*}
    R_k = \underbrace{\sum_{j=-m}^{m} w_j \sum_{i=1}^{N_p} p_i^{(k)} \cdot f(\text{pixel}_{j,i})}_{\text{Naive: }(2m{+}1) \times N_p\text{ reads}}
    \;=\; \underbrace{\sum_{\ell=1}^{N'_k} \alpha_{k,\ell} \cdot f(\text{pixel}_\ell)}_{\text{Fused: }N'_k\text{ reads}}
  \end{equation*}

  \vspace{0.3em}
  \begin{itemize}
    \item Many $(j, i)$ pairs map to the \textbf{same integer pixel} after rounding
    \item Sum their weights $\;\to\;$ single precomputed $\alpha_{k,\ell}$ per unique position
    \item The entire multi-step pipeline collapses:
  \end{itemize}

  \vspace{0.3em}
  \begin{center}
    {\small
    \color{UofSCRose}Rotate $\to$ Build $\mathbf{A}$ $\to$ $\mathbf{A}^+$ $\to$ Gather $\to$ MatMul $\to$ Weight $\to$ Sum}\\[0.4em]
    $\Downarrow$\\[0.4em]
    {\color{UofSCHorseshoe}\textbf{Single Gather\; + \;Dot Product}\; with precomputed $\boldsymbol{\alpha}_k$}
  \end{center}

  \vspace{0.3em}
  \begin{exampleblock}{Result}
    $R_k = \boldsymbol{\alpha}_k^\top \mathbf{g}_k$ \quad---\quad
    \textbf{zero runtime matrix operations}.
  \end{exampleblock}
\end{frame}

\begin{frame}{Stencil Shapes Across Orientations}
  \begin{center}
    \includegraphics[width=0.78\textwidth]{fig3_stencil_orientations.png}
  \end{center}

  \vspace{0.3em}
  \begin{itemize}
    \item Each dot is one unique pixel in the fused stencil ($m{=}7$, $N_p{=}15$, $d{=}4$)
    \item Color encodes the fused weight $\alpha$ (red = positive, blue = negative)
    \item Dipole pattern rotates with $\theta$; stencil elongates along the candidate edge
    \item Arrow shows the normal derivative direction $f_x$
  \end{itemize}
\end{frame}

\sectionsubtitle{Per-pixel cost after stencil fusion}
\section{ASF Operation Count}

\begin{frame}[fragile]{Fused ASF: Pseudocode}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \begin{lstlisting}[language=Python, escapeinside={(*}{*)}]
# ONE-TIME precompute:
#   (*$\boldsymbol{\alpha}_k$*) for each orientation

for (*$\theta$*) in orientations:   # (*$N_s$*) = 18
  for i = 1 to (*$N'$*):         # (*$N'$*) (*$\approx$*) 264
    READ pixel at offset[i] # 1 read
    MUL by (*$\alpha$*)[i]        # 1 multiply
  SUM (*$N'$*) values             # 263 adds
  ABS((*$R_\theta$*))
pick max over (*$\theta$*)
      \end{lstlisting}
    \end{column}
    \begin{column}{0.45\textwidth}
      \textbf{Per orientation} ($m{=}7,\ N_p{=}100$):
      \begin{itemize}
        \item Memory reads: \textbf{264}
        \item Multiplies: \textbf{264}
        \item Additions: \textbf{263}
      \end{itemize}

      \vspace{0.6em}
      \textbf{Per pixel} (18 orientations):
      \begin{itemize}
        \item Reads: \textbf{4,752}
        \item Multiplies: \textbf{4,752}
        \item Additions: \textbf{4,734}
        \item[\color{UofSCHorseshoe}$\blacksquare$] \textbf{Total: $\sim$14,238 ops}
      \end{itemize}
    \end{column}
  \end{columns}

  \vspace{0.3em}
  \begin{block}{Complexity}
    $\mathcal{O}\!\bigl(N_s \times N'\bigr)$ per pixel \qquad where $N' \approx 0.18 \times (2m{+}1) \times N_p$
  \end{block}
\end{frame}

\begin{frame}{Where Do the Savings Come From?}
  \begin{columns}[T]
    \begin{column}{0.31\textwidth}
      \begin{block}{1.\ Deduplication}
        \begin{itemize}
          \item Overlapping neighborhoods share pixels
          \item 82\% reduction at $m{=}7$
          \item $1{,}500 \to 264$ reads
          \item Improves with $m$
        \end{itemize}
        \vspace{0.3em}
        $\to$ \textbf{5.7$\times$} fewer memory accesses
      \end{block}
    \end{column}
    \begin{column}{0.31\textwidth}
      \begin{block}{2.\ Precomputed Weights}
        \begin{itemize}
          \item Pseudoinverse, rotation, Gaussian all fused into $\boldsymbol{\alpha}_k$
          \item \textbf{Zero} runtime matrix ops
          \item Eliminates $\mathcal{O}(N_p {\times} M)$ matmul/orient.
        \end{itemize}
        \vspace{0.3em}
        $\to$ Removes entire matmul stage
      \end{block}
    \end{column}
    \begin{column}{0.31\textwidth}
      \begin{block}{3.\ GPU Kernel Design}
        \begin{itemize}
          \item Single gather + dot per orientation
          \item Vectorized across 128-px blocks
          \item Coalesced row reads
        \end{itemize}
        \vspace{0.3em}
        $\to$ Constant \textbf{20~MB} VRAM, fits any GPU
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\sectionsubtitle{}
\section{Summary}

\begin{frame}{Summary}
  \textbf{The naive line filter is memory-bound and redundant:}
  \begin{itemize}
    \item $\mathcal{O}(N_s \times L \times N_p)$ per pixel, most pixel reads duplicated
    \item $\sim$81,000 total ops/pixel at $m{=}7$, $N_p{=}100$, $N_s{=}18$
    \item 6.2~GB VRAM for a single 720p image
  \end{itemize}

  \vspace{0.6em}
  \textbf{The fused stencil eliminates all redundancy:}
  \begin{itemize}
    \item $\mathcal{O}(N_s \times N')$ per pixel, $N' \approx 0.18 \times L \times N_p$
    \item $\sim$14,000 total ops/pixel --- \textbf{5.7$\times$ arithmetic reduction}
    \item \textbf{Zero runtime matrix operations}
  \end{itemize}

  \vspace{0.6em}
  \textbf{GPU results exceed the theoretical arithmetic speedup:}
  \begin{itemize}
    \item \textbf{18.4$\times$} wall-clock at $m{=}7$, \textbf{24$\times$} at $m{=}14$
    \item \textbf{311$\times$} VRAM reduction (6.2~GB $\to$ 20~MB)
    \item Enables deployment on \emph{any} GPU, including embedded
  \end{itemize}
\end{frame}
